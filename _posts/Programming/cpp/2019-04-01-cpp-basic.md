#include <iostream>
using namespace std;

template<typename T>
class Test
{
    typedef T typeT;
	using usingT = T;

    typeT v1 = 1;
	usingT v2 = 2;
public:

    void foo()
    {
        if (false)
            return;

        for (int i = 0; i < 10; ++i)
        {
            cout << "foo" << endl;
        }
    }

	void bar() const
	{
		cout << "bar" << endl;
	}
};
int main()
{
    Test<int> ttt;
    ttt.foo();
    ttt.bar();
}
